66 void SndFileInstrument::setParameter(unsigned selector, int argc, void **argv, const char *types) {
67  float d = * (float *) argv[0];
68  if (types[0] == 'i')
69  d = (float) (* (int *) argv[0]);
70  switch (selector) {
